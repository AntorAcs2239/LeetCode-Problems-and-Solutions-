class Solution {
public:
static bool cmp(string a,string b)
{
    return a+b>b+a;
}
    string largestNumber(vector<int>& ar) {
        vector<string>vec;
        for(int i=0;i<ar.size();i++)
        {
            vec.push_back(to_string(ar[i]));
        }
        sort(vec.begin(),vec.end(),cmp);
        string ans;
        for(auto x:vec)ans+=x;
        return ans;
    }
};
